def filter_patients_by_specialty(patients, selected_specialties, specialty_keywords):
    filtered = []
    for patient in patients:
        patient_specialties = set()
        patient_reports = MedicalReport.query.filter_by(patient_id=patient.id).all()
        
        for report in patient_reports:
            conditions = report.get_conditions()
            for condition in conditions:
                condition_lower = condition.lower()
                for specialty, keywords in specialty_keywords.items():
                    for keyword in keywords:
                        if keyword.lower() in condition_lower:
                            patient_specialties.add(specialty)
                            break
        
        if selected_specialties:
            if patient_specialties.intersection(set(selected_specialties)):
                filtered.append(patient)
        else:
            filtered.append(patient)
    
    return filtered

// ===== دالة تحليل التقارير المبسطة =====
async function analyzeReports(imageFiles, patientId = null) {
    const status = document.getElementById("analysisStatus");
    const resultBox = document.getElementById("analysisResult");
    const loading = document.getElementById("loading");
    const resultsSection = document.getElementById("resultsSection");
    
    if (!imageFiles?.length) {
        alert("Please select at least one file");
        return;
    }

    // إظهار حالة التحميل
    status.style.display = "block";
    status.textContent = `📤 Processing ${imageFiles.length} image(s)...`;
    status.className = "status-message info";
    if (resultBox) resultBox.style.display = "none";
    if (loading) loading.classList.remove("hidden");
    if (resultsSection) resultsSection.classList.add("hidden");

    let allText = "";
    let successCount = 0;
    let imagePaths = [];

    // معالجة كل صورة
    for (let i = 0; i < imageFiles.length; i++) {
        const file = imageFiles[i];
        status.textContent = `📤 Processing ${i+1}/${imageFiles.length}: ${file.name}`;
        
        const formData = new FormData();
        formData.append("file", file);

        try {
            const response = await fetch("/ocr", { method: "POST", body: formData });
            const data = await response.json();
            
            if (data.error) throw new Error(data.error);
            
            allText += `\n--- Image ${i+1} ---\n${data.text}\n`;
            if (data.image_path) imagePaths.push(data.image_path);
            successCount++;
        } catch (error) {
            console.error(`Error: ${file.name}`, error);
        }
    }

    if (successCount === 0) {
        status.innerHTML = "❌ No text could be extracted. Please ensure images are clear.";
        status.className = "status-message error";
        if (loading) loading.classList.add("hidden");
        return;
    }

    // تحليل AI
    status.textContent = `🔍 AI analyzing ${successCount} image(s)...`;

    const prompt = `Extract medical info from this report. Return ONLY valid JSON:
{
    "conditions": [],
    "medications": [],
    "alerts": [],
    "summary": ""
}

Text: ${allText.substring(0, 3000)}`;

    try {
        const aiResponse = await puter.ai.chat(prompt);
        
        // استخراج JSON من الرد
        let content = typeof aiResponse === 'string' ? aiResponse : (aiResponse.content || aiResponse.message?.content || JSON.stringify(aiResponse));
        let cleanContent = content.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim();
        let analysis = JSON.parse(cleanContent);
        
        // التأكد من وجود الحقول
        analysis.conditions = analysis.conditions || [];
        analysis.medications = analysis.medications || [];
        analysis.alerts = analysis.alerts || [];
        analysis.summary = analysis.summary || "Analysis complete";

        // عرض النتائج في الجدول
        const tableBody = document.getElementById("resultTableBody");
        if (tableBody) {
            const date = new Date().toLocaleDateString();
            tableBody.innerHTML = `
                <tr class="border-b hover:bg-gray-50">
                    <td class="px-4 py-3">${date}</td>
                    <td class="px-4 py-3">${analysis.conditions.map(c => `• ${escape(c)}`).join('<br>') || '-'}</td>
                    <td class="px-4 py-3">${analysis.medications.map(m => `• ${escape(m)}`).join('<br>') || '-'}</td>
                    <td class="px-4 py-3">${analysis.alerts.map(a => `⚠️ ${escape(a)}`).join('<br>') || '-'}</td>
                    <td class="px-4 py-3">${escape(analysis.summary)}</td>
                </tr>
            `;
        }

        // حفظ للمستخدمين المسجلين فقط
        if (patientId && imagePaths.length) {
            await fetch("/report/save-report", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    patient_id: patientId,
                    image_paths: imagePaths,
                    conditions: analysis.conditions,
                    medications: analysis.medications,
                    alerts: analysis.alerts,
                    summary: analysis.summary
                })
            });
        }

        // إظهار النتائج
        status.innerHTML = "✅ Analysis complete!";
        status.className = "status-message success";
        if (resultBox) resultBox.style.display = "block";
        if (resultsSection) resultsSection.classList.remove("hidden");

    } catch (error) {
        console.error(error);
        status.innerHTML = "❌ Analysis failed. Please try again.";
        status.className = "status-message error";
    } finally {
        if (loading) loading.classList.add("hidden");
        setTimeout(() => { if (status) status.style.display = "none"; }, 3000);
    }
}

function escape(text) {
    if (!text) return '';
    const div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
}

document.addEventListener("DOMContentLoaded", function() {
    const form = document.getElementById("quickAnalysisForm");
    const fileInput = document.getElementById("quickReportImage");
    const preview = document.getElementById("imagePreview");
    const analyzeBtn = document.getElementById("analyzeBtn");
    
    let selectedFiles = [];
    
    // اختيار الملفات
    document.getElementById("selectFilesBtn")?.addEventListener("click", () => fileInput.click());
    
    // السحب والإفلات
    const dropzone = document.getElementById("dropzone");
    if (dropzone) {
        dropzone.addEventListener("click", () => fileInput.click());
        dropzone.addEventListener("dragover", (e) => { e.preventDefault(); dropzone.classList.add("border-blue-500", "bg-blue-50"); });
        dropzone.addEventListener("dragleave", () => dropzone.classList.remove("border-blue-500", "bg-blue-50"));
        dropzone.addEventListener("drop", (e) => {
            e.preventDefault();
            dropzone.classList.remove("border-blue-500", "bg-blue-50");
            handleFiles(Array.from(e.dataTransfer.files));
        });
    }
    
    fileInput?.addEventListener("change", (e) => handleFiles(Array.from(e.target.files)));
    
    function handleFiles(files) {
        for (const file of files) {
            if (!file.type.startsWith("image/")) continue;
            selectedFiles.push(file);
            
            const reader = new FileReader();
            reader.onload = (e) => {
                const div = document.createElement("div");
                div.className = "relative";
                div.innerHTML = `
                    <img src="${e.target.result}" class="w-full h-24 object-cover rounded-lg">
                    <button type="button" onclick="this.parentElement.remove(); removeFile('${file.name}')" class="absolute top-1 right-1 bg-red-500 text-white rounded-full w-6 h-6 text-xs">×</button>
                `;
                preview.appendChild(div);
            };
            reader.readAsDataURL(file);
        }
        analyzeBtn.disabled = selectedFiles.length === 0;
        updateFileInput();
    }
    
    window.removeFile = (name) => {
        selectedFiles = selectedFiles.filter(f => f.name !== name);
        analyzeBtn.disabled = selectedFiles.length === 0;
        updateFileInput();
    };
    
    function updateFileInput() {
        const dt = new DataTransfer();
        selectedFiles.forEach(f => dt.items.add(f));
        fileInput.files = dt.files;
    }
    
    // إرسال النموذج
    form?.addEventListener("submit", (e) => {
        e.preventDefault();
        if (selectedFiles.length) analyzeReports(selectedFiles, null);
        else alert("Please select images");
    });
});

    // Section switching
    function showSection(section) {
        document.getElementById('profileSection').style.display = 'none';
        document.getElementById('passwordSection').style.display = 'none';
        document.getElementById('preferencesSection').style.display = 'none';
        {% if session.get('user_role') == 'patient' %}

        document.getElementById(section + 'Section').style.display = 'block';
        
        // Update active button style
        const buttons = ['btnProfile', 'btnPassword', 'btnPreferences', 'btnDoctors'];
        buttons.forEach(btn => {
            const el = document.getElementById(btn);
            if (el) {
                el.classList.remove('bg-blue-100', 'text-blue-600');
                el.classList.add('hover:bg-blue-50');
            }
        });
        const activeBtn = document.getElementById('btn' + section.charAt(0).toUpperCase() + section.slice(1));
        if (activeBtn) {
            activeBtn.classList.add('bg-blue-100', 'text-blue-600');
            activeBtn.classList.remove('hover:bg-blue-50');
        }

        fetch('/set-language/' + lang).then(() => location.reload());
    }
    
    function showDeleteModal() {
        document.getElementById('deleteModal').style.display = 'flex';
    }
    
    function closeDeleteModal() {
        document.getElementById('deleteModal').style.display = 'none';
    }

    document.getElementById('deleteModal')?.addEventListener('click', function(e) {
        if (e.target === this) closeDeleteModal();
    });
    
    // Password validation
    const newPass = document.getElementById('newPassword');
    const confirmPass = document.getElementById('confirmPassword');
    if (newPass && confirmPass) {
        confirmPass.addEventListener('input', function() {
            if (newPass.value !== this.value) {
                this.setCustomValidity('{{ gettext("Passwords do not match") }}');
            } else {
                this.setCustomValidity('');
            }
        });
    }

    function uploadLogo(input) {
        if (input.files && input.files[0]) {
            const formData = new FormData();
            formData.append('logo', input.files[0]);
            
            fetch('/upload-logo', {
                method: 'POST',
                body: formData
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    location.reload();
                } else {
                    alert('Error: ' + data.error);
                }
            })
            .catch(error => console.error('Error:', error));
        }
    }

    function removeLogo() {
        if (confirm('{{ gettext("Are you sure you want to remove the clinic logo?") }}')) {
            fetch('/remove-logo', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' }
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    location.reload();
                } else {
                    alert('Error: ' + data.error);
                }
            })
            .catch(error => console.error('Error:', error));
        }
    }

<script>
    // Toggle Password Visibility
    const togglePassword = document.getElementById('togglePassword');
    const passwordInput = document.getElementById('passwordInput');
    
    if (togglePassword) {
        togglePassword.addEventListener('click', function() {
            const type = passwordInput.getAttribute('type') === 'password' ? 'text' : 'password';
            passwordInput.setAttribute('type', type);
            this.querySelector('i').classList.toggle('fa-eye');
            this.querySelector('i').classList.toggle('fa-eye-slash');
        });
    }
    
    // Toggle between Patient and Doctor login
    const patientBtn = document.getElementById('patientBtn');
    const doctorBtn = document.getElementById('doctorBtn');
    const ssnField = document.getElementById('ssnField');
    const emailField = document.getElementById('emailField');
    const submitBtn = document.getElementById('submitBtn');
    const userType = document.getElementById('userType');
    const ssnInput = document.getElementById('ssnInput');
    const emailInput = document.getElementById('emailInput');
    
    function setLoginType(type) {
        if (type === 'patient') {
            // Patient mode
            patientBtn.classList.remove('bg-gray-200', 'text-gray-600');
            patientBtn.classList.add('bg-blue-600', 'text-white');
            doctorBtn.classList.remove('bg-blue-600', 'text-white');
            doctorBtn.classList.add('bg-gray-200', 'text-gray-600');
            
            ssnField.style.display = 'block';
            emailField.style.display = 'none';
            submitBtn.innerHTML = '<i class="fas fa-sign-in-alt mr-2"></i>{{ gettext("Login as Patient") }}';
            userType.value = 'patient';
            
            if (ssnInput) ssnInput.required = true;
            if (emailInput) emailInput.required = false;
            
        } else {
            // Doctor mode
            doctorBtn.classList.remove('bg-gray-200', 'text-gray-600');
            doctorBtn.classList.add('bg-blue-600', 'text-white');
            patientBtn.classList.remove('bg-blue-600', 'text-white');
            patientBtn.classList.add('bg-gray-200', 'text-gray-600');
            
            ssnField.style.display = 'none';
            emailField.style.display = 'block';
            submitBtn.innerHTML = '<i class="fas fa-sign-in-alt mr-2"></i>{{ gettext("Login as Doctor") }}';
            userType.value = 'doctor';
            
            if (ssnInput) ssnInput.required = false;
            if (emailInput) emailInput.required = true;
        }
    }
    
    if (patientBtn) {
        patientBtn.addEventListener('click', () => setLoginType('patient'));
    }
    
    if (doctorBtn) {
        doctorBtn.addEventListener('click', () => setLoginType('doctor'));
    }
    
    // Form validation
    const form = document.getElementById('loginForm');
    if (form) {
        form.addEventListener('submit', function(e) {
            const type = userType.value;
            
            if (type === 'patient') {
                const ssn = ssnInput?.value.trim();
                if (!ssn) {
                    e.preventDefault();
                    alert('{{ gettext("Please enter your SSN") }}');
                    return false;
                }
            } else {
                const email = emailInput?.value.trim();
                if (!email) {
                    e.preventDefault();
                    alert('{{ gettext("Please enter your email") }}');
                    return false;
                }
            }
            
            const password = passwordInput?.value;
            if (!password) {
                e.preventDefault();
                alert('{{ gettext("Please enter your password") }}');
                return false;
            }
            
            return true;
        });
    }
</script>

<script>
    // Toggle Password Visibility
    const togglePassword = document.getElementById('togglePassword');
    const passwordInput = document.getElementById('passwordInput');
    
    if (togglePassword) {
        togglePassword.addEventListener('click', function() {
            const type = passwordInput.getAttribute('type') === 'password' ? 'text' : 'password';
            passwordInput.setAttribute('type', type);
            this.querySelector('i').classList.toggle('fa-eye');
            this.querySelector('i').classList.toggle('fa-eye-slash');
        });
    }
    
    // Role Selection Toggle
    const roleSelect = document.getElementById('roleSelect');
    const doctorFields = document.getElementById('doctorFields');
    const patientFields = document.getElementById('patientFields');
    
    function toggleRoleFields() {
        if (roleSelect.value === 'doctor') {
            doctorFields.classList.remove('hidden');
            patientFields.classList.add('hidden');
        } else {
            doctorFields.classList.add('hidden');
            patientFields.classList.remove('hidden');
        }
    }
    
    roleSelect.addEventListener('change', toggleRoleFields);
    toggleRoleFields();
    
    // Find Doctors for Patients
    const findBtn = document.getElementById('findDoctorsBtn');
    const communeSelect = document.getElementById('patientCommuneSelect');
    const specialtySelect = document.getElementById('patientSpecialtySelect');
    const doctorsContainer = document.getElementById('doctorsListContainer');
    const doctorsList = document.getElementById('doctorsList');
    const doctorIdsInput = document.getElementById('doctorIdsInput');
    
    let selectedDoctors = [];
    
    if (findBtn) {
        findBtn.addEventListener('click', async function() {
            const communeId = communeSelect.value;
            const specialtyId = specialtySelect.value;
            
            if (!communeId || !specialtyId) {
                alert('{{ gettext("Please select both commune and specialty") }}');
                return;
            }
            
            try {
                const response = await fetch(`/auth/get-doctors?commune_id=${communeId}&specialty_id=${specialtyId}`);
                const data = await response.json();
                
                doctorsList.innerHTML = '';
                
                if (data.doctors.length === 0) {
                    doctorsList.innerHTML = '<div class="text-center text-gray-500 py-4">{{ gettext("No doctors found") }}</div>';
                } else {
                    data.doctors.forEach(doc => {
                        const isChecked = selectedDoctors.includes(doc.id);
                        const div = document.createElement('div');
                        div.className = 'flex items-center p-2 hover:bg-gray-50 rounded border';
                        div.innerHTML = `
                            <input type="checkbox" value="${doc.id}" id="doc_${doc.id}" class="mr-3 doctor-checkbox" ${isChecked ? 'checked' : ''}>
                            <label for="doc_${doc.id}" class="flex-1 cursor-pointer">
                                <div class="font-medium">${doc.name}</div>
                                <div class="text-sm text-gray-500">${doc.specialty}</div>
                            </label>
                            <i class="fas fa-phone-alt text-gray-400 text-sm"> ${doc.phone}</i>
                        `;
                        doctorsList.appendChild(div);
                    });
                    
                    // Add event listeners to checkboxes
                    document.querySelectorAll('.doctor-checkbox').forEach(cb => {
                        cb.addEventListener('change', function() {
                            const id = parseInt(this.value);
                            if (this.checked) {
                                if (!selectedDoctors.includes(id)) selectedDoctors.push(id);
                            } else {
                                selectedDoctors = selectedDoctors.filter(i => i !== id);
                            }
                            doctorIdsInput.value = selectedDoctors.join(',');
                        });
                    });
                }
                
                doctorsContainer.classList.remove('hidden');
                
            } catch (error) {
                console.error('Error:', error);
                alert('{{ gettext("Error finding doctors") }}');
            }
        });
    }
    
    // Form Validation
    const form = document.getElementById('registerForm');
    form.addEventListener('submit', function(e) {
        if (roleSelect.value === 'patient') {
            const ssn = document.getElementById('ssnInput').value;
            if (!ssn || ssn.trim() === '') {
                e.preventDefault();
                alert('{{ gettext("SSN is required for patient registration") }}');
                return false;
            }
        }
        return true;
    });
</script>

<script>
    // Timer functionality (10 minutes = 600 seconds)
    let timeLeft = 600;
    const timerElement = document.getElementById('timer');
    const verifyBtn = document.getElementById('verifyBtn');
    const resendLink = document.getElementById('resendLink');
    
    function formatTime(seconds) {
        const mins = Math.floor(seconds / 60);
        const secs = seconds % 60;
        return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
    }
    
    function updateTimer() {
        if (timeLeft <= 0) {
            timerElement.textContent = '00:00';
            timerElement.classList.add('text-red-500');
            verifyBtn.disabled = true;
            verifyBtn.classList.add('opacity-50', 'cursor-not-allowed');
            resendLink.classList.add('text-blue-600', 'hover:underline');
            return;
        }
        timerElement.textContent = formatTime(timeLeft);
        timeLeft--;
        setTimeout(updateTimer, 1000);
    }
    
    updateTimer();
</script>

    // Toggle Account Fields
    const createAccountCheckbox = document.getElementById('createAccountCheckbox');
    const accountFields = document.getElementById('accountFields');
    const ssnInput = document.getElementById('ssnInput');
    const emailInput = document.getElementById('emailInput');
    const accountEmail = document.getElementById('accountEmail');
    const passwordInput = document.getElementById('passwordInput');
    
    if (createAccountCheckbox) {
        createAccountCheckbox.addEventListener('change', function() {
            if (this.checked) {
                accountFields.classList.remove('hidden');
                ssnInput.required = true;
                passwordInput.required = true;
                if (emailInput.value) {
                    accountEmail.value = emailInput.value;
                }
            } else {
                accountFields.classList.add('hidden');
                ssnInput.required = false;
                passwordInput.required = false;
                accountEmail.value = '';
            }
        });
    }
    
    // Copy email from main field to account email
    if (emailInput && accountEmail) {
        emailInput.addEventListener('input', function() {
            if (createAccountCheckbox.checked && !accountEmail.value) {
                accountEmail.value = this.value;
            }
        });
    }
    
    // Toggle Password Visibility
    const togglePassword = document.getElementById('togglePassword');
    if (togglePassword) {
        togglePassword.addEventListener('click', function() {
            const type = passwordInput.getAttribute('type') === 'password' ? 'text' : 'password';
            passwordInput.setAttribute('type', type);
            this.querySelector('i').classList.toggle('fa-eye');
            this.querySelector('i').classList.toggle('fa-eye-slash');
        });
    }
    
    // Form Validation
    const form = document.getElementById('addPatientForm');
    form.addEventListener('submit', function(e) {
        const isCreatingAccount = createAccountCheckbox.checked;
        
        if (isCreatingAccount) {
            const ssn = ssnInput.value.trim();
            const password = passwordInput.value;
            
            if (!ssn) {
                e.preventDefault();
                alert('{{ gettext("SSN is required when creating an account") }}');
                return false;
            }
            
            if (!password || password.length < 6) {
                e.preventDefault();
                alert('{{ gettext("Password must be at least 6 characters") }}');
                return false;
            }
        }
        
        return true;
    });

<script>
    // ==================== DOM Elements ====================
    const dropzone = document.getElementById('dropzone');
    const fileInput = document.getElementById('quickReportImage');
    const selectFilesBtn = document.getElementById('selectFilesBtn');
    const imagePreview = document.getElementById('imagePreview');
    const analyzeBtn = document.getElementById('analyzeBtn');
    const loading = document.getElementById('loading');
    const analysisStatus = document.getElementById('analysisStatus');
    const resultTableBody = document.getElementById('resultTableBody');
    const actionButtons = document.getElementById('actionButtons');
    
    let selectedFiles = [];
    
    // ==================== Helper Functions ====================
    function escapeHtml(text) {
        if (!text) return '';
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }
    
    // ==================== analyzeReports Function ====================
    async function analyzeReports(imageFiles, patientId = null) {
        if (!imageFiles || imageFiles.length === 0) {
            alert("{{ gettext('Please select at least one file') }}");
            return;
        }

        analysisStatus.style.display = "block";
        analysisStatus.textContent = `📤 Processing ${imageFiles.length} image(s)...`;
        analysisStatus.className = "status-message info";
        
        let allText = "";
        let successCount = 0;
        let imagePaths = [];

        for (let i = 0; i < imageFiles.length; i++) {
            const file = imageFiles[i];
            
            analysisStatus.textContent = `📤 Processing image ${i+1} of ${imageFiles.length}: ${file.name}`;
            
            const formData = new FormData();
            formData.append("file", file);

            try {
                const ocrResponse = await fetch("/ocr", {
                    method: "POST",
                    body: formData
                });

                if (!ocrResponse.ok) {
                    console.error(`OCR failed for ${file.name}`);
                    continue;
                }

                const ocrData = await ocrResponse.json();
                allText += `\n\n--- Page ${i+1}: ${file.name} ---\n${ocrData.text}\n`;
                imagePaths.push(ocrData.image_path);
                successCount++;
                
            } catch (error) {
                console.error(`Error processing ${file.name}:`, error);
            }
        }

        if (successCount === 0) {
            analysisStatus.textContent = "❌ No images could be processed";
            analysisStatus.className = "status-message error";
            return;
        }

        analysisStatus.textContent = `🔍 Analyzing ${successCount} images with AI...`;

        const prompt = `Extract medical information from this complete medical report and return JSON.

Medical Report:
${allText.substring(0, 4000)}

Return exactly this format:
{
    "conditions": [],
    "medications": [],
    "alerts": [],
    "summary": ""
}`;

        try {
            const aiResponse = await puter.ai.chat(prompt);
            
            let analysis;
            try {
                let content;
                if (aiResponse.message && aiResponse.message.content) {
                    content = aiResponse.message.content;
                } else if (aiResponse.content) {
                    content = aiResponse.content;
                } else if (typeof aiResponse === 'string') {
                    content = aiResponse;
                } else {
                    content = JSON.stringify(aiResponse);
                }
                
                let cleanContent = content.trim();
                if (cleanContent.startsWith('```json')) {
                    cleanContent = cleanContent.replace(/```json\n?/g, '').replace(/```\n?/g, '');
                } else if (cleanContent.startsWith('```')) {
                    cleanContent = cleanContent.replace(/```\n?/g, '');
                }
                
                analysis = JSON.parse(cleanContent);
                analysis.conditions = analysis.conditions || [];
                analysis.medications = analysis.medications || [];
                analysis.alerts = analysis.alerts || [];
                analysis.summary = analysis.summary || "No summary provided";
                
            } catch (e) {
                console.log("Parse error, using fallback:", e);
                analysis = { 
                    summary: "Analysis completed. Please consult a doctor for accurate interpretation.",
                    conditions: [],
                    medications: [],
                    alerts: []
                };
            }

            // Display results in table
            const currentDate = new Date().toLocaleDateString('en-US', {
                year: 'numeric',
                month: 'short',
                day: 'numeric'
            });
            
            const conditionsList = analysis.conditions?.length > 0
                ? `<ul class="list-disc list-inside">${analysis.conditions.map(c => `<li>${escapeHtml(c)}</li>`).join('')}</ul>`
                : '<span class="text-gray-400">{{ gettext("None identified") }}</span>';
            
            const medicationsList = analysis.medications?.length > 0
                ? `<ul class="list-disc list-inside">${analysis.medications.map(m => `<li>${escapeHtml(m)}</li>`).join('')}</ul>`
                : '<span class="text-gray-400">{{ gettext("None identified") }}</span>';
            
            const alertsList = analysis.alerts?.length > 0
                ? `<ul class="list-disc list-inside">${analysis.alerts.map(a => `<li class="text-yellow-700">⚠️ ${escapeHtml(a)}</li>`).join('')}</ul>`
                : '<span class="text-green-600"><i class="fas fa-check-circle"></i> {{ gettext("No alerts") }}</span>';
            
            resultTableBody.innerHTML = `
                <tr class="border-b hover:bg-gray-50">
                    <td class="px-3 py-3 font-medium">${currentDate}</td>
                    <td class="px-3 py-3">${conditionsList}</td>
                    <td class="px-3 py-3">${medicationsList}</td>
                    <td class="px-3 py-3">${alertsList}</td>
                    <td class="px-3 py-3">${escapeHtml(analysis.summary)}</td>
                </tr>
            `;

            analysisStatus.textContent = "✅ Analysis complete!";
            analysisStatus.className = "status-message success";

            // Save to database if patientId is provided
            if (patientId && imagePaths.length > 0) {
                try {
                    const saveResponse = await fetch("/report/save-report", {
                        method: "POST",
                        headers: { "Content-Type": "application/json" },
                        body: JSON.stringify({
                            patient_id: patientId,
                            image_paths: imagePaths,
                            conditions: analysis.conditions || [],
                            medications: analysis.medications || [],
                            alerts: analysis.alerts || [],
                            summary: analysis.summary || "Analysis complete"
                        })
                    });
                    
                    if (saveResponse.ok) {
                        console.log("Report saved successfully");
                    }
                } catch(e) {
                    console.error("Save error:", e);
                }
            }
            
            // Show action buttons
            actionButtons.style.display = 'flex';
            
            // Store for save button
            window.lastAnalysis = analysis;
            window.lastImagePaths = imagePaths;
            
            return analysis;

        } catch (error) {
            console.error("Error:", error);
            analysisStatus.textContent = "❌ Error: " + error.message;
            analysisStatus.className = "status-message error";
        }
    }
    
    // ==================== File Handling Functions ====================
    function handleFiles(files) {
        for (const file of files) {
            if (!file.type.startsWith('image/')) continue;
            
            selectedFiles.push(file);
            
            const reader = new FileReader();
            reader.onload = (e) => {
                const div = document.createElement('div');
                div.className = 'image-preview';
                div.id = `preview_${file.name.replace(/[^a-zA-Z0-9]/g, '_')}`;
                div.innerHTML = `
                    <img src="${e.target.result}" alt="${file.name}">
                    <button type="button" class="remove-image" onclick="removeImage('${file.name.replace(/'/g, "\\'")}')">
                        <i class="fas fa-times"></i>
                    </button>
                `;
                imagePreview.appendChild(div);
            };
            reader.readAsDataURL(file);
        }
        analyzeBtn.disabled = selectedFiles.length === 0;
        
        // Update file input
        const dataTransfer = new DataTransfer();
        selectedFiles.forEach(f => dataTransfer.items.add(f));
        fileInput.files = dataTransfer.files;
    }
    
    window.removeImage = (name) => {
        selectedFiles = selectedFiles.filter(f => f.name !== name);
        const previewDiv = document.getElementById(`preview_${name.replace(/[^a-zA-Z0-9]/g, '_')}`);
        if (previewDiv) previewDiv.remove();
        
        // Update file input
        const dataTransfer = new DataTransfer();
        selectedFiles.forEach(f => dataTransfer.items.add(f));
        fileInput.files = dataTransfer.files;
        
        analyzeBtn.disabled = selectedFiles.length === 0;
    };
    
    // ==================== Event Listeners ====================
    dropzone.addEventListener('click', () => fileInput.click());
    selectFilesBtn.addEventListener('click', () => fileInput.click());
    
    dropzone.addEventListener('dragover', (e) => {
        e.preventDefault();
        dropzone.classList.add('border-blue-500', 'bg-blue-50');
    });
    
    dropzone.addEventListener('dragleave', () => {
        dropzone.classList.remove('border-blue-500', 'bg-blue-50');
    });
    
    dropzone.addEventListener('drop', (e) => {
        e.preventDefault();
        dropzone.classList.remove('border-blue-500', 'bg-blue-50');
        handleFiles(Array.from(e.dataTransfer.files));
    });
    
    fileInput.addEventListener('change', (e) => {
        handleFiles(Array.from(e.target.files));
    });
    
    // Analyze button click
    analyzeBtn.addEventListener('click', async () => {
        if (selectedFiles.length === 0) {
            alert('{{ gettext("Please select images to analyze") }}');
            return;
        }
        
        // Reset UI
        loading.classList.remove('hidden');
        analysisStatus.style.display = 'none';
        resultTableBody.innerHTML = '';
        actionButtons.style.display = 'none';
        analyzeBtn.disabled = true;
        
        try {

    async function checkDrugInteractions(patientId, selectedMedications) {
        const alertsContainer = document.getElementById('aiAlerts');
        const alertsContent = document.getElementById('aiAlertsContent');
        
        if (!selectedMedications.length) {
            if (alertsContainer) alertsContainer.style.display = 'none';
            return;
        }
        
        try {
            // عرض حالة التحميل
            if (alertsContainer) {
                alertsContainer.style.display = 'block';
                alertsContent.innerHTML = '<div class="text-center py-4"><i class="fas fa-spinner fa-spin"></i> {{ gettext("AI analyzing drug interactions...") }}</div>';
            }
            
            // 1. جلب الأدوية السابقة للمريض من API
            const reportsResponse = await fetch(`/prescription/api/patient/${patientId}/reports`);
            const reports = await reportsResponse.json();
            
            let previousMedications = [];
            reports.forEach(report => {
                if (report.medications && report.medications.length > 0) {
                    previousMedications.push(...report.medications);
                }
            });
            
            // 2. تجهيز أسماء الأدوية الجديدة
            const newMedsList = selectedMedications.map(m => m.name).join(', ');
            const oldMedsList = previousMedications.slice(0, 10).join(', ') || 'None';
            
            // 3. إنشاء الـ Prompt لـ Puter AI
            const prompt = `You are a medical AI assistant. Analyze drug interactions between current medications and new prescription.

    Current medications: ${oldMedsList}
    New medications: ${newMedsList}

    Respond with ONLY valid JSON in this format (no other text):
    {
        "interactions": [
            {
                "description": "short description of interaction",
                "severity": "high/medium/low",
                "recommendation": "short recommendation"
            }
        ],
        "summary": "one sentence summary",
        "safe_to_prescribe": true/false
    }

    If no interactions found, return empty array for interactions.`;

            // 4. استدعاء Puter AI مباشرة من Frontend
            const aiResponse = await puter.ai.chat(prompt);
            
            // 5. معالجة الرد
            let result;
            try {
                let content = aiResponse;
                if (aiResponse.message && aiResponse.message.content) {
                    content = aiResponse.message.content;
                } else if (aiResponse.content) {
                    content = aiResponse.content;
                }
                
                // تنظيف النص من علامات JSON
                let cleanContent = content.trim();
                if (cleanContent.startsWith('```json')) {
                    cleanContent = cleanContent.replace(/```json\n?/g, '').replace(/```\n?/g, '');
                } else if (cleanContent.startsWith('```')) {
                    cleanContent = cleanContent.replace(/```\n?/g, '');
                }
                
                result = JSON.parse(cleanContent);
            } catch(e) {
                console.error("Parse error:", e);
                result = { interactions: [], summary: "Analysis complete", safe_to_prescribe: true };
            }
            
            // 6. عرض النتائج
            if (alertsContainer && alertsContent) {
                if (result.interactions && result.interactions.length > 0) {
                    let alertsHtml = '';
                    result.interactions.forEach(i => {
                        const severityClass = i.severity === 'high' ? 'text-red-600 border-red-300 bg-red-50' : 
                                            (i.severity === 'medium' ? 'text-yellow-600 border-yellow-300 bg-yellow-50' : 
                                            'text-blue-600 border-blue-300 bg-blue-50');
                        alertsHtml += `
                            <div class="mb-3 p-3 border rounded-lg ${severityClass}">
                                <p class="font-medium"><i class="fas fa-exclamation-triangle"></i> ${escapeHtml(i.description)}</p>
                                <p class="text-sm text-gray-600 mt-1"><i class="fas fa-lightbulb"></i> ${escapeHtml(i.recommendation)}</p>
                            </div>
                        `;
                    });
                    alertsHtml += `<div class="mt-3 pt-3 border-t"><strong>{{ gettext('Summary') }}:</strong> ${escapeHtml(result.summary)}</div>`;
                    if (!result.safe_to_prescribe) {
                        alertsHtml += `<div class="mt-2 p-2 bg-red-100 text-red-700 rounded"><i class="fas fa-ban"></i> {{ gettext('This prescription may not be safe! Please review.') }}</div>`;
                    }
                    alertsContent.innerHTML = alertsHtml;
                } else {
                    alertsContent.innerHTML = '<div class="text-green-600"><i class="fas fa-check-circle"></i> {{ gettext("No dangerous drug interactions detected") }}</div>';
                }
            }
            
            return result;
            
        } catch (error) {
            console.error("AI Error:", error);
            if (alertsContainer && alertsContent) {
                alertsContent.innerHTML = '<div class="text-yellow-600"><i class="fas fa-exclamation-triangle"></i> {{ gettext("AI service unavailable. Please check interactions manually.") }}</div>';
            }
            return { interactions: [], safe_to_prescribe: true };
        }
    }

    function escapeHtml(text) {

    async function analyzeDosageSafety(patientId, selectedMedications) {
    const dosageDiv = document.getElementById('dosageAnalysis');
    const tableBody = document.getElementById('dosageTableBody');
    
    if (!selectedMedications.length) {
        if (dosageDiv) dosageDiv.style.display = 'none';
        return;
    }
    
    try {
        dosageDiv.style.display = 'block';
        tableBody.innerHTML = '<tr><td colspan="5" class="text-center py-4"><i class="fas fa-spinner fa-spin"></i> Analyzing...</td></tr>';
        
        let allIngredients = [];
        
        // جلب المواد الفعالة لكل دواء
        for (let med of selectedMedications) {
            try {
                // استخدم المسار الصحيح للـ API
                const response = await fetch(`/prescription/api/medicine/${med.medicine_id}/ingredients`);
                
                if (!response.ok) {
                    console.error(`Failed to fetch ingredients for medicine ${med.medicine_id}`);
                    continue;
                }
                
                const data = await response.json();
                
                // التحقق من وجود ingredients في الرد
                if (data.ingredients && data.ingredients.length > 0) {
                    allIngredients.push(...data.ingredients);
                } else if (data.length > 0) {
                    // إذا كان الرد مصفوفة مباشرة
                    allIngredients.push(...data);
                }
            } catch(e) {
                console.error(`Error fetching ingredients for ${med.name}:`, e);
            }
        }
        
        // تجميع المواد الفعالة المكررة
        const grouped = {};
        allIngredients.forEach(ing => {
            const ingName = ing.name || ing.ingredient_name;
            if (!grouped[ingName]) {
                grouped[ingName] = {
                    name: ingName,
                    total_daily_dose: 0,
                    max_daily_dose_mg: ing.max_daily_dose_mg || 2000,
                    warning_message: ing.warning_message || 'Exceeds recommended daily dose'
                };
            }
            // تقدير الجرعة اليومية (مرتين في اليوم)
            grouped[ingName].total_daily_dose += (ing.amount_mg || 500) * 2;
        });
        
        if (Object.keys(grouped).length === 0) {
            tableBody.innerHTML = '<tr><td colspan="5" class="text-center py-4 text-gray-400">No active ingredients found - Dosage analysis unavailable</td></tr>';
            return;
        }
        
        // عرض النتائج
        let html = '';
        let hasDanger = false;
        let hasWarning = false;
        
        for (const [name, data] of Object.entries(grouped)) {
            const isSafe = data.total_daily_dose <= data.max_daily_dose_mg;
            let statusClass = 'text-green-600';
            let statusText = '✓ Safe';
            let bgClass = '';
            
            if (!isSafe) {
                if (data.total_daily_dose > data.max_daily_dose_mg * 1.5) {
                    statusClass = 'text-red-600';
                    statusText = '🔴 Exceeds limit (High)';
                    bgClass = 'bg-red-50';
                    hasDanger = true;
                } else {
                    statusClass = 'text-yellow-600';
                    statusText = '⚠️ Exceeds limit';
                    bgClass = 'bg-yellow-50';
                    hasWarning = true;
                }
            }
            
            html += `
                <tr class="border-b ${bgClass}">
                    <td class="px-3 py-2 font-medium">${escapeHtml(name)}</td>
                    <td class="px-3 py-2">${data.total_daily_dose} mg</td>
                    <td class="px-3 py-2">${data.max_daily_dose_mg} mg</td>
                    <td class="px-3 py-2"><span class="${statusClass} font-medium">${statusText}</span></td>
                    <td class="px-3 py-2">${!isSafe ? (data.warning_message || '⚠️ Exceeds safe daily limit') : '—'}</td>
                </tr>
            `;
        }
        
        // إضافة ملخص
        let summaryHtml = '';
        if (hasDanger) {
            summaryHtml = '<div class="mb-3 p-2 bg-red-100 text-red-700 rounded"><i class="fas fa-exclamation-triangle"></i> Some medications exceed safe dosage limits significantly!</div>';
        } else if (hasWarning) {
            summaryHtml = '<div class="mb-3 p-2 bg-yellow-100 text-yellow-700 rounded"><i class="fas fa-exclamation-circle"></i> Some medications are at the upper limit of safe dosage.</div>';
        } else {
            summaryHtml = '<div class="mb-3 p-2 bg-green-100 text-green-700 rounded"><i class="fas fa-check-circle"></i> All medications are within safe dosage limits.</div>';
        }
        
        const summaryDiv = document.getElementById('dosageSummary');
        if (summaryDiv) summaryDiv.innerHTML = summaryHtml;
        
        tableBody.innerHTML = html;
        
    } catch (error) {
        console.error("Error analyzing dosage:", error);
        tableBody.innerHTML = '<tr><td colspan="5" class="text-center py-4 text-red-500">Error analyzing dosages. Please check if medicine data is available.</td></tr>';
    }
}

async function onMedicationsChange(patientId) {
        const medicineSelects = document.querySelectorAll('.medicine-select');
        const dosageSelects = document.querySelectorAll('.dosage-select');
        const frequencySelects = document.querySelectorAll('.frequency-select');
        const durationSelects = document.querySelectorAll('.duration-select');
        
        const selectedMedications = [];
        for (let i = 0; i < medicineSelects.length; i++) {
            if (medicineSelects[i].value) {
                selectedMedications.push({
                    medicine_id: medicineSelects[i].value,
                    name: medicineSelects[i].options[medicineSelects[i].selectedIndex]?.text.split(' (')[0] || '',
                    dosage: dosageSelects[i]?.value || '',
                    frequency: frequencySelects[i]?.value || '',
                    duration: durationSelects[i]?.value || ''
                });
            }
        }
        
        if (selectedMedications.length > 0 && patientId) {
            await checkDrugInteractions(patientId, selectedMedications);
            await analyzeDosageSafety(patientId, selectedMedications);
        } else {
            document.getElementById('aiAlerts').style.display = 'none';
            document.getElementById('dosageAnalysis').style.display = 'none';
        }
    }

<script>
    async function viewReport(reportId) {
        const modal = document.getElementById('reportModal');
        const modalContent = document.getElementById('modalContent');
        
        modal.style.display = 'flex';
        modalContent.innerHTML = '<div class="text-center py-8"><div class="loader mx-auto"></div><p class="mt-3">{{ gettext("Loading...") }}</p></div>';
        
        try {
            const response = await fetch(`/report/${reportId}/export`);
            const report = await response.json();
            
            const conditionsList = report.conditions?.length > 0
                ? `<ul class="list-disc list-inside">${report.conditions.map(c => `<li>${escapeHtml(c)}</li>`).join('')}</ul>`
                : '<span class="text-gray-400">-</span>';
            
            const medicationsList = report.medications?.length > 0
                ? `<ul class="list-disc list-inside">${report.medications.map(m => `<li>${escapeHtml(m)}</li>`).join('')}</ul>`
                : '<span class="text-gray-400">-</span>';
            
            const alertsList = report.alerts?.length > 0
                ? `<ul class="list-disc list-inside">${report.alerts.map(a => `<li class="text-yellow-700">⚠️ ${escapeHtml(a)}</li>`).join('')}</ul>`
                : '<span class="text-green-600">✓ {{ gettext("No alerts") }}</span>';
            
            modalContent.innerHTML = `
                <div class="space-y-4">
                    <div class="grid grid-cols-2 gap-3 pb-3 border-b">
                        <div><p class="text-sm text-gray-500">{{ gettext('Date') }}</p><p class="font-medium">${new Date(report.report_date).toLocaleDateString()}</p></div>
                        <div><p class="text-sm text-gray-500">{{ gettext('Report ID') }}</p><p class="font-medium">#${report.id}</p></div>
                    </div>
                    <div><h4 class="font-semibold mb-2"><i class="fas fa-stethoscope text-blue-500 mr-2"></i>{{ gettext('Conditions') }}</h4><div class="bg-blue-50 p-3 rounded-lg">${conditionsList}</div></div>
                    <div><h4 class="font-semibold mb-2"><i class="fas fa-pills text-blue-500 mr-2"></i>{{ gettext('Medications') }}</h4><div class="bg-green-50 p-3 rounded-lg">${medicationsList}</div></div>
                    <div><h4 class="font-semibold mb-2"><i class="fas fa-exclamation-triangle text-blue-500 mr-2"></i>{{ gettext('Alerts') }}</h4><div class="bg-yellow-50 p-3 rounded-lg">${alertsList}</div></div>
                    <div><h4 class="font-semibold mb-2"><i class="fas fa-align-left text-blue-500 mr-2"></i>{{ gettext('Summary') }}</h4><div class="bg-gray-50 p-3 rounded-lg">${escapeHtml(report.summary) || '-'}</div></div>
                    <div class="pt-3 border-t flex justify-end gap-2">
                        <button onclick="closeModal()" class="px-4 py-2 bg-gray-500 text-white rounded-lg hover:bg-gray-600">{{ gettext('Close') }}</button>
                        <button onclick="window.print()" class="px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700"><i class="fas fa-print mr-2"></i>{{ gettext('Print') }}</button>
                    </div>
                </div>
            `;
        } catch (error) {
            modalContent.innerHTML = '<div class="text-center py-8 text-red-500"><i class="fas fa-exclamation-circle text-3xl mb-2"></i><p>{{ gettext("Error loading report") }}</p></div>';
        }
    }
    
    function closeModal() {
        document.getElementById('reportModal').style.display = 'none';
    }
    
    function escapeHtml(text) {
        if (!text) return '';
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }
    
    document.getElementById('reportModal')?.addEventListener('click', function(e) {
        if (e.target === this) closeModal();
    });
</script>

<script>
    let currentPrescriptionId = null;
    
    function sharePrescription(prescriptionId) {
        currentPrescriptionId = prescriptionId;
        const modal = document.getElementById('shareModal');
        modal.style.display = 'flex';
        document.getElementById('shareEmail').value = '';
    }
    
    function closeShareModal() {
        const modal = document.getElementById('shareModal');
        modal.style.display = 'none';
        currentPrescriptionId = null;
    }
    
    async function sendShare() {
        const email = document.getElementById('shareEmail').value.trim();
        if (!email) {
            alert('{{ gettext("Please enter an email address") }}');
            return;
        }
        
        if (!email.includes('@') || !email.includes('.')) {
            alert('{{ gettext("Please enter a valid email address") }}');
            return;
        }
        
        const btn = event.target;
        const originalText = btn.innerHTML;
        btn.innerHTML = '<i class="fas fa-spinner fa-spin mr-2"></i>{{ gettext("Sending...") }}';
        btn.disabled = true;
        
        try {
            const response = await fetch('/prescription/share', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    prescription_id: currentPrescriptionId,
                    email: email
                })
            });
            
            const data = await response.json();
            
            if (response.ok) {
                alert('{{ gettext("Prescription sent successfully!") }}');
                closeShareModal();
            } else {
                alert(data.error || '{{ gettext("Error sending prescription") }}');
            }
        } catch (error) {
            console.error('Error:', error);
            alert('{{ gettext("Error sending prescription. Please try again.") }}');
        } finally {
            btn.innerHTML = originalText;
            btn.disabled = false;
        }
    }
    
    // Close modal when clicking outside
    document.getElementById('shareModal')?.addEventListener('click', function(e) {
        if (e.target === this) closeShareModal();
    });
</script>
{% endblock %}

# ===== إعداد الترجمة =====
def get_current_language():
    """الحصول على اللغة الحالية من الجلسة"""
    return session.get('language', 'en')

def gettext(text):
    """دالة الترجمة للاستخدام في القوالب"""
    try:
        from backend.translate import translations
        lang = get_current_language()
        # دعم الترجمة المتعددة
        if isinstance(translations.get(lang, {}), dict):
            return translations.get(lang, translations.get('en', {})).get(text, text)
        return text
    except (ImportError, AttributeError):
        # إذا لم يوجد ملف translate.py، أرجع النص الأصلي
        return text

# إضافة دوال الترجمة إلى بيئة القوالب
app.jinja_env.globals.update(
    gettext=gettext,
    _=gettext,  # اختصار
    current_lang=get_current_language
)

# ===== مسار تغيير اللغة (مع دعم POST) =====
@app.route('/set-language/<lang>')
def set_language(lang):
    if lang in ['ar', 'en']:
        session['language'] = lang
        session.permanent = True
    return redirect(request.referrer or url_for('home'))

@app.route('/set-language', methods=['POST'])
def set_language_post():
    """API endpoint لتغيير اللغة عبر JavaScript"""
    data = request.get_json()
    lang = data.get('lang', 'en')
    if lang in ['ar', 'en']:
        session['language'] = lang
        return jsonify({'success': True, 'language': lang})
    return jsonify({'success': False}), 400
